\section*{Symbols}
\begin{itemize}
\item[$F$] impact force [N]
\item[$m_{arm}$] effective mass of the arm [kg]
\item[$\mu_{gender}$] anthropometric arm mass coefficient [\%]
\item[$\vec{a}$] linear acceleration vector
\item[$\vec{v}_{aim}$] normalized aiming direction vector
\item[$P_{cue}$] center position of the cue ball
\item[$P_{ghost}$] calculated position of the ghost ball
\item[$L_{proj}$] length of the projection vector
\item[$d_{\perp}$] perpendicular distance to the aiming line
\end{itemize}
